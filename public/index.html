/ GET

/apps GET
List all apps
/apps/create POST
Creates a new app, saves to mongodb
body = {
    name: string,
    originator: string,
    collaborators: string[],
    features: string[],
    timeline: string[]
}
/apps/initiate POST
Updates the apps initiated state to true
body = {
    _id: string
}
/features/save POST
Appends a new feature to an existing apps feature list
body = {
    _id: string,
    feature: string
}
/features/delete DELETE
Deletes a feature from an existing apps feature list
body = {
    _id: string
}
